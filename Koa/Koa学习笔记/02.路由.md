# 路由

## 原生路由实现

通过判断或正则匹配 `ctx.request.url` ，就可以制定出所需的路由。

```js
const Koa = require('koa')
const fs = require('fs')
const app = new Koa()

// 用 promise 封装异步读取文件方法
function render(page) {
  return new Promise((resolve, reject) => {
    let viewUrl = `./view/${page}`
    fs.readFile(viewUrl, 'binary', (err, data) => {
      if (err) {
        reject(err)
      } else {
        resolve(data)
      }
    })
  })
}

// 根据 RUL 读取 HTML 文件
async function route(url) {
  let view = '404.html'
  switch(url) {
    case '/':
      view = 'index.html'
      break
    case '/index':
      view = 'index.html'
      break
    case '/todo':
      view = 'todo.html'
      break
    case '/404':
      view = '404.html'
      break
    default:
      break
  }
  let html = await render(view)
  return html
}

app.use(async (ctx) => {
  // 获取当前访问路径
  let url = ctx.request.url
  let html = await route(url)
  ctx.body = html
})

// 监听端口
app.listen(8787, () => console.log('http server is runing at port 8787'))
```

## koa-router 中间件

如果依靠 `ctx.request.url` 去手动处理路由，会写很多处理代码。可以使用 `koa-router` 中间件。

```js
const Koa = require('koa')
const Router = require('koa-router')
const app = new Koa()

// 父路由
let router = new Router()
router
  .get('/', async (ctx) => {
    ctx.body = 'home'
  })
  .get('/todo', async (ctx) => {
    ctx.body = 'home / todo'
  })
  .get('/product', async (ctx) => {
    ctx.body = 'home / product'
  })
  .get('/404', async (ctx) => {
    ctx.body = '404 page'
  })

// 子路由 1
let todo = new Router()
todo
  .get('/list', async (ctx) => {
    ctx.body = 'todo / list'
  })
  .get('/detail', async (ctx) => {
    ctx.body = 'todo / detail'
  })

// 子路由 2
let product = new Router()
product
  .get('/list', async (ctx) => {
    ctx.body = 'product / list'
  })
  .get('/detail', async (ctx) => {
    ctx.body = 'product / detail'
  })

// 装载子路由到父级路由
router.use('/todo', todo.routes(), todo.allowedMethods())
router.use('/product', product.routes(), product.allowedMethods())

// 装载路由中间件
app
  .use(router.routes())
  .use(router.allowedMethods())

app.listen(8787, () => console.log('http server is runing at port 8787'))
```
